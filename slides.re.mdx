import { defineHandler, type InferProps } from "void";
import { db } from "void/db";
import { users } from "@schema";

# What is <Lowercase>_fate_</Lowercase>?

- A modern data client inspired by Relay and GraphQL
- View composition
- Normalized cache
- Data masking
- Pagination, live views, and optimistic updates

→ _[fate.technology](https://fate.technology)_

# Defining Views

<div style={{fontSize: '0.62em', alignSelf: 'center'}}>

```ts {4,10}
import type { Post, User } from "@app/fate/views";
import { view } from "react-fate";

export const UserView = view<User>()({
  id: true,
  name: true,
  username: true,
});

export const PostView = view<Post>()({
  author: UserView,
  id: true,
  title: true,
});
```

# Reading Views

import { PostView, UserView } from "../src/fateViews";

function PostCard({ post: postRef }: { post: ViewRef<"Post"> }) {
  const post = useView(PostView, postRef);
  const author = useView(UserView, post.author);

  return (

function LikeButton({ post }: { post: Post }) {

import { useListView, useRequest } from "vue-fate";
import { PostConnectionView } from "../src/fateViews";

const { posts: postsRef } = await useRequest({
  posts: { list: PostConnectionView },
}).ready();
const [posts, loadNext] = useListView(PostConnectionView, postsRef);
</script>

<template>
  <PostCard v-for="{ node } in posts" :key="node.id" :post="node" />

# Full Stack

- SSR calls the fate server directly
- Browser requests go to `/fate`
- Live updates go through `/fate-live`
- Void handles the runtime, database, live stream, and deploy
- _fate_ owns component data, cache, pagination, live views, and mutations
